1. Add to Cart Callback

Called when user clicks "Add to Cart" on a product recommendation.

addToCartCallback: (product) => {
  console.log('Add to cart:', product);
  
  // Product object structure:
  // {
  //   id: 'prod_123',
  //   title: 'Laptop Pro',
  //   description: 'High-performance laptop',
  //   price: 1299.99,
  //   currency: 'USD',
  //   imageUrl: 'https://...',
  //   url: 'https://...',
  //   inStock: true,
  //   metadata: { sku: 'LAP-001', category: 'Electronics' }
  // }
  
  // Example: Add to cart via API
  fetch('/api/cart/add', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      productId: product.id,
      quantity: 1,
      price: product.price
    })
  })
  .then(response => response.json())
  .then(data => {
    // Show success notification
    showNotification(`Added ${product.title} to cart!`);
    
    // Update cart count in UI
    updateCartCount(data.cartCount);
  })
  .catch(error => {
    console.error('Failed to add to cart:', error);
    showNotification('Failed to add to cart', 'error');
  });
}

3. Analytics Callback

Called for various analytics events throughout the conversation.

analyticsCallback: (event) => {
  console.log('Analytics event:', event);
  
  // Event object structure:
  // {
  //   event: 'chat_query' | 'product_view' | 'product_click' | 'add_to_cart',
  //   timestamp: '2025-11-02T10:30:00Z',
  //   sessionId: 'sess_abc123',
  //   query: 'Show me laptops',
  //   responseTime: 450,
  //   resultsCount: 3,
  //   metadata: { ... }
  // }
  
  // Example: Track with Google Analytics
  if (typeof gtag !== 'undefined') {
    gtag('event', event.event, {
      event_category: 'RAG Assistant',
      event_label: event.query,
      value: event.responseTime,
      session_id: event.sessionId
    });
  }
  
  // Example: Track with Segment
  if (typeof analytics !== 'undefined') {
    analytics.track(event.event, {
      query: event.query,
      responseTime: event.responseTime,
      resultsCount: event.resultsCount,
      sessionId: event.sessionId
    });
  }
  
  // Example: Track with Mixpanel
  if (typeof mixpanel !== 'undefined') {
    mixpanel.track(event.event, {
      query: event.query,
      responseTime: event.responseTime,
      resultsCount: event.resultsCount
    });
  }
  
  // Example: Send to your own analytics API
  fetch('/api/analytics/track', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(event)
  });
}

4. Error Callback

Called when an error occurs in the widget.

errorCallback: (error) => {
  console.error('Widget error:', error);
  
  // Error object structure:
  // {
  //   type: 'api_error' | 'network_error' | 'validation_error',
  //   message: 'Failed to fetch response',
  //   code: 'ERR_NETWORK',
  //   timestamp: '2025-11-02T10:30:00Z',
  //   details: { ... }
  // }
  
  // Example: Show user-friendly error message
  if (error.type === 'network_error') {
    showNotification('Connection lost. Please check your internet.', 'error');
  } else if (error.type === 'api_error') {
    showNotification('Service temporarily unavailable. Please try again.', 'error');
  }
  
  // Example: Log to error tracking service
  if (typeof Sentry !== 'undefined') {
    Sentry.captureException(new Error(error.message), {
      tags: {
        component: 'rag-assistant',
        errorType: error.type
      },
      extra: error.details
    });
  }
}

Called when user sends a message.

onMessageSent: (message) => {

  // Message object structure:
  // {
  //   text: 'Show me laptops under $1000',

  trackEngagement('message_sent', {
    messageLength: message.text.length,
    sessionId: message.sessionId
  });
}

6. Message Received Callback

Called when assistant responds with a message.

onMessageReceived: (message) => {
  console.log('Message received:', message);
  
  // Message object structure:
  // {
  //   text: 'Here are some laptops under $1000...',
  //   timestamp: '2025-11-02T10:30:01Z',
  //   sessionId: 'sess_abc123',
  //   products: [...],
  //   responseTime: 450
  // }
  
  // Example: Track response quality
  trackEngagement('message_received', {
    responseTime: message.responseTime,
    hasProducts: message.products.length > 0,
    sessionId: message.sessionId
  });
}

Complete Example

Here's a complete example with all callbacks:

const assistant = new RAGAssistant({
  merchantId: 'your_merchant_id',
  apiKey: 'pk_live_...',
  
  integration: {
    // Cart integration
    addToCartCallback: (product) => {
      addToCart(product);
      showNotification(`Added ${product.title} to cart!`);
      trackEvent('add_to_cart', { productId: product.id });
    },
    
    // Checkout integration
    checkoutCallback: (items) => {
      window.location.href = '/checkout';
      trackEvent('checkout_initiated', { itemCount: items.length });
    },
    
    // Analytics integration
    analyticsCallback: (event) => {
      // Google Analytics
      gtag('event', event.event, {
        event_category: 'RAG Assistant',
        event_label: event.query,
        value: event.responseTime
      });
      
      // Custom analytics
      trackEvent(event.event, event);
    },
    
    // Error handling
    errorCallback: (error) => {
      console.error('Widget error:', error);
      showNotification('Something went wrong. Please try again.', 'error');
      
      // Log to Sentry
      Sentry.captureException(new Error(error.message));
    },
    
    // Message events
    onMessageSent: (message) => {
      trackEvent('message_sent', { messageLength: message.text.length });
    },
    
    onMessageReceived: (message) => {
      trackEvent('message_received', { 
        responseTime: message.responseTime,
        hasProducts: message.products.length > 0
      });
    },
    
    // Widget state events
    onOpen: () => {
      trackEvent('widget_opened');
      pauseBackgroundMedia();
    },
    
    onClose: () => {
      trackEvent('widget_closed');
      resumeBackgroundMedia();
    }
  }
});
