Shopify

Most popular e-commerce platform

Installation Steps

  1. Go to your Shopify admin panel
  2. Navigate to Online Store ‚Üí Themes
  3. Click Actions ‚Üí Edit code
  4. Open theme.liquid
  5. Paste the code below before </body>

Code Snippet

React / Vue / Angular

Modern JavaScript frameworks

Installation Steps

  1. Install via NPM: npm install @rag-assistant/widget
  2. Import in your component
  3. Initialize in componentDidMount or useEffect

React Example

import { useEffect } from 'react';
import RAGAssistant from '@rag-assistant/widget';

function App() {
  useEffect(() => {
    const assistant = new RAGAssistant({
      merchantId: 'YOUR_MERCHANT_ID',
      apiKey: 'YOUR_API_KEY',
      integration: {
        addToCartCallback: (product) => {
          // Use your Redux/Context cart actions
          dispatch(addToCart(product));
        }
      }
    });
    
    return () => {
      // Cleanup if needed
    };
  }, []);
  
  return <div>Your App</div>;
}

Vue Example

import RAGAssistant from '@rag-assistant/widget';

export default {
  mounted() {
    this.assistant = new RAGAssistant({
      merchantId: 'YOUR_MERCHANT_ID',
      apiKey: 'YOUR_API_KEY',
      integration: {
        addToCartCallback: (product) => {
          // Use Vuex store
          this.$store.dispatch('addToCart', product);
        }
      }
    });
  }
}
