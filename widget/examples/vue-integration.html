Method 2: Vue 3 Options API

Using the traditional Options API.

<template>
  <div id="app">
    <h1>My Vue Store</h1>
  </div>
</template>

<script>
import RAGAssistant from '@rag-assistant/widget';

export default {
  name: 'App',
  
  data() {
    return {
      assistant: null
    };
  },
  
  mounted() {
    this.assistant = new RAGAssistant({
      merchantId: 'your_merchant_id',
      apiKey: 'pk_live_...',
      
      theme: {
        primaryColor: '#42b883',
        position: 'bottom-right'
      },
      
      integration: {
        addToCartCallback: (product) => {
          // Call component method
          this.addToCart(product);
        },
        
        checkoutCallback: (items) => {
          // Navigate using Vue Router
          this.$router.push('/checkout');
        }
      }
    });
  },
  
  methods: {
    addToCart(product) {
      // Your cart logic
      console.log('Add to cart:', product);
    },
    
    openChat() {
      if (this.assistant) {
        this.assistant.open();
      }
    }
  },
  
  beforeUnmount() {
    // Cleanup
    this.assistant = null;
  }
};

Method 3: With Vuex Store

Integrate with Vuex for state management.

// store/index.js
import { createStore } from 'vuex';

export default createStore({
  state: {
    cart: []
  },
  
  mutations: {
    ADD_TO_CART(state, product) {
      state.cart.push(product);
    }
  },
  
  actions: {
    addToCart({ commit }, product) {
      commit('ADD_TO_CART', product);
    }
  }
});

// App.vue
<script setup>
import { onMounted } from 'vue';
import { useStore } from 'vuex';
import { useRouter } from 'vue-router';
import RAGAssistant from '@rag-assistant/widget';

const store = useStore();
const router = useRouter();

onMounted(() => {
  const assistant = new RAGAssistant({
    merchantId: 'your_merchant_id',
    apiKey: 'pk_live_...',
    
    integration: {
      addToCartCallback: (product) => {
        // Dispatch Vuex action
        store.dispatch('addToCart', product);
        
        // Show notification
        ElMessage.success(`Added ${product.title} to cart!`);
      },
      
      checkoutCallback: (items) => {
        // Navigate to checkout
        router.push('/checkout');
      },
      
      analyticsCallback: (event) => {
        // Track with Google Analytics
        if (window.gtag) {
          window.gtag('event', event.event, {
            event_category: 'RAG Assistant',
            event_label: event.query
          });
        }
      }
    }
  });
});
</script>

Method 5: As a Vue Plugin

Create a reusable Vue plugin for the RAG Assistant.

// plugins/rag-assistant.js
import RAGAssistant from '@rag-assistant/widget';

export default {
  install(app, options) {
    const assistant = new RAGAssistant(options);
    
    // Make available globally
    app.config.globalProperties.$ragAssistant = assistant;
    
    // Provide/inject pattern
    app.provide('ragAssistant', assistant);
  }
};

// main.js
import { createApp } from 'vue';
import App from './App.vue';
import RAGAssistantPlugin from './plugins/rag-assistant';

const app = createApp(App);

app.use(RAGAssistantPlugin, {
  merchantId: 'your_merchant_id',
  apiKey: 'pk_live_...',
  theme: {
    primaryColor: '#42b883'
  }
});

app.mount('#app');

// Usage in components:
<script setup>
import { inject } from 'vue';

const ragAssistant = inject('ragAssistant');

const openChat = () => {
  ragAssistant.open();
};
</script>

// Or with Options API:
export default {
  methods: {
    openChat() {
      this.$ragAssistant.open();
    }
  }
};

Method 6: Composable (Vue 3)

Create a reusable composable for the RAG Assistant.

// composables/useRAGAssistant.js
import { onMounted, onBeforeUnmount, ref } from 'vue';
import RAGAssistant from '@rag-assistant/widget';

export function useRAGAssistant(config) {
  const assistant = ref(null);
  const isReady = ref(false);
  
  onMounted(() => {
    assistant.value = new RAGAssistant(config);
    isReady.value = true;
  });
  
  onBeforeUnmount(() => {
    assistant.value = null;
    isReady.value = false;
  });
  
  const open = () => {
    if (assistant.value) {
      assistant.value.open();
    }
  };
  
  const close = () => {
    if (assistant.value) {
      assistant.value.close();
    }
  };
  
  const sendMessage = (message) => {
    if (assistant.value) {
      assistant.value.sendMessage(message);
    }
  };
  
  return {
    assistant,
    isReady,
    open,
    close,
    sendMessage
  };
}

// Usage in component:
<script setup>
import { useRAGAssistant } from './composables/useRAGAssistant';
import { useCartStore } from './stores/cart';

const cartStore = useCartStore();

const { open, sendMessage } = useRAGAssistant({
  merchantId: 'your_merchant_id',
  apiKey: 'pk_live_...',
  integration: {
    addToCartCallback: (product) => {
      cartStore.addItem(product);
    }
  }
});
</script>
ðŸ’¡ Pro Tips:
  • Use onMounted to initialize the widget after DOM is ready
  • Store the assistant instance in component data or ref
  • Integrate with Vuex or Pinia for state management
  • Create a plugin or composable for reusability
  • Use onBeforeUnmount for cleanup
  • Leverage Vue Router for navigation in callbacks
